  • 如果同時是 3 和 5 的倍數，就印出 “FizzBuzz”
  • 將結果回傳為一個陣列。

        function fizzBuzz100() {
            let result = []
            let count = 1
            while (count <= 100) {
                if ((count % 3 === 0) && (count % 5 === 0)) {
                    result.push("“FizzBuzz”");
                } else if ((count % 3 === 0)) {
                    result.push("“Buzz”");
                } else if ((count % 5 === 0)) {
                    result.push("Fizz");
                } else {
                    result.push(count.toString());
                }
                count++
            }
            console.log(result);
            return result
        }
