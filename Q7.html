題目七：字串去左右空白

// input 輸入
trimString("    hi, mom!  ")
trimString(" aa@gmail.com  ")

// output 輸出
"hi, mom!"
"aa@gmail.com"

回答:

        function trimString(str) {
            str = str.trim();
            if (str.length > 0) {//不是NaN時才運算              
                return str.trim();
            } else {
                console.log('無帶入文字');
            }
        }      
        console.log(trimString("    hi, mom!  "));
        console.log(trimString(" aa@gmail.com  "));
