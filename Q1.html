// input 輸入
multiply3Times(3);
multiply3Times(0);
multiply3Times(9);

// output 輸出
27
0
729

回答:

      function multiply3Times(num) {
            if (!isNaN(num)) {//不是NaN時才運算              
                console.log(num * num * num);
            } else {
                console.log(num + '-數字有問題無法運算');
            }
        }
        multiply3Times(3);
        multiply3Times(0);
        multiply3Times(9);
        multiply3Times('哈');
