第七題：乘法表

請製作一個可以打印 a x b 大小的乘法表的函式：

✨keywords：nested for loops

// Input 輸入
console.log(tableAxB(9, 9))

// Output 輸出
"
1 * 1 = 1
1 * 2 = 2
1 * 3 = 3
1 * 4 = 4
1 * 5 = 5
1 * 6 = 6
1 * 7 = 7
1 * 8 = 8
1 * 9 = 9
//
//... 省略 ...
//
9 * 1 = 9
9 * 2 = 18
9 * 3 = 27
9 * 4 = 36
9 * 5 = 45
9 * 6 = 54
9 * 7 = 63
9 * 8 = 72
9 * 9 = 81
"

回答:

        function tableAxB(x, y) {
            let answer = "";
            for (let i = 1; i <= x; i++) {
                for (let j = 1; j <= y; j++) {
                    answer += `${i} * ${j} = ${i * j}\n`;
                }
            }
            return answer;
        }
