第四題：反轉陣列

請製作一個函式可以回傳反轉後的陣列。

✨keywords：for

 // Todo 待處理
function reverseArray(arr) {
    //...
}
console.log(reverseArray([1,2,3,4,5]))

// Output 輸出
[5,4,3,2,1]

回答:

        function reverseArray(arr) {
            let answer = [];
            //前面都是正序，這邊以反序的方式做迴圈，將起始值作為 length-1 ，因為索引值是從 0 開始，所以下方陣列最大的索引值為 4 ，最小值為 0 ，接著就能從後向前迴圈並 push 到陣列重新排序。
            for (let i = arr.length - 1; i >= 0; i--) {
                answer.push(arr[i]);
            }
            return answer;
        }

        // function reverseArray(arr) {
        //    return arr.reverse();
        // }
        
        console.log(reverseArray([1, 2, 3, 4, 5]))

        // Output 輸出
        //[5, 4, 3, 2, 1]
