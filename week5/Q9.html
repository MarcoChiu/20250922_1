第九題：蓋房子

請製作一個函式，輸入一正整數（n）回傳一棟用文字構成的方磚屋（n*n），規則如下：

  • 房子上方正中央為「門」，不能放中間的話將門往任一邊旁邊放一格
  • n = 3, x門x
  • n = 4 -> xx門x 或 x門xx
  • 替整個房子加上屋頂骨架（對角線是「骨」）
  • 房子內其他剩餘空間由「空」構成

回答:

    function square(num) {
      let str = "";
      for (let i = 0; i < num; i++) {
        for (let j = 0; j < num; j++) {

          //這個判斷式先將兩塊的磚畫出來
          if (num <= 2) {
            str += "磚";
          }

          //接著這是三塊以上的部分
          //在寫這類題目前可以思考它的規則
          else if (num >= 3) {

            //這是從左上到右下的"骨"線，它的位置 i == j
            if (i == j) {
              str += "骨";
            }

            //這是右上到左下的"骨"線，它的位置會等於 i+j == num-1 eg:[0,4][1,3][2,2][3,1][4,0]
            else if (num - 1 == i + j) {
              str += "骨";
            }

            //這是寫出門的位置，會使用 Math.floor() 找出若不可被2整除時，其最大整數的值作為開門位置
            else if (i == 0 && j == Math.floor(num / 2)) {
              str += "門";
            }

            //寫出四面牆的位置
            else if (i == 0 || i == num - 1 || j == 0 || j == num - 1) {
              str += "磚";
            }

            //最後填上空格
            else {
              str += "空";
            };
          };
        };
        str += "\n";
      };
      return str;
    };
    // Input 輸入
    console.log(square(1))
    console.log(square(2))
    console.log(square(3))
    console.log(square(4))
    console.log(square(5))
