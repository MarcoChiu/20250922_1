回答:

        function countArray(arr) {
            let answer = {};
            for (let i = 0; i < arr.length; i++) {
                //若 answer 沒有的 arr[i] ， arr[i] 會給予一個起始值建立資料。
                if (!answer[arr[i]]) {
                    answer[arr[i]] = 1;

                    //否則會在有的資料內增加一筆。
                } else {
                    answer[arr[i]]++;
                }
            }
            return answer;
        }
