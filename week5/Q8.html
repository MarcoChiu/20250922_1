第八題：蓋金字塔

請輸入金字塔的最高峰並使用 console.log 印出結果

✨keywords：for

// Input 輸入
pyramid(5);

// Output 輸出
"*"
"**"
"***"
"****"
"*****"
"****"
"***"
"**"
"*"

回答:

        function pyramid(num) {
            //跑迴圈時，會將 i 傳入 star() 的函式，下方函式就能依據傳入的參數花星號。
            for (let i = 1; i <= num; i++) {
                console.log(star(i));
            }

            //反序
            for (let i = num - 1; i > 0; i--) {
                console.log(star(i));
            }
        }

        //這裡是畫星字號的部分。
        function star(i) {
            let result = "";
            for (let j = 1; j <= i; j++) {
                result += "*";
            }
            return result;
        }
