回答:

        const group = [{
            name: "小明",
            age: 18
        },
        {
            name: "小美",
            age: 7
        },
        {
            name: "小華",
            age: 80
        }];
        function filterUnder18(persons) {
            let answer = [];
            //以 for ... of 取得其值，接下來使用判斷式判斷是否滿18，未滿18的將其 push 入 array 。
            for (let i of group) {
                console.log(i.age);
                if (i.age < 18) {
                    answer.push(i);
                }
            }
            return answer;
        }
