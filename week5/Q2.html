        const seasonFruit = {
            春天: ["梅子", "枇杷"],
            夏天: ["西瓜", "桃子"],
            秋天: ["火龍果", "百香果"],
            冬天: ["橘子", "柳丁"]
        };

        function printObject(obj) {
            let answer = '';
            //使用 for ... in 取其 key ，接著以 key 值取得其值，最後將答案 push 入 array 後 return 出來呈現答案。
            //key:春天,value:["梅子", "枇杷"]
            for (let i in obj) {
                //這裡的 \n 是一種名為轉譯字符的形式，會在編譯時呈現與字面上不同的意思(可以理解成暗號)， \n 即是換行的意思。
                //console.log(i);
                answer += `${i}有${obj[i].join("、")}\n`;
            }
            return answer;
        }

        //     const mArr = Object.entries(obj);

        //         console.log(`${el[0]}有${el[1][0]}、${el[1][1]}`);
        //     });
        // }
