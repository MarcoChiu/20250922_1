回答:

        function getMaxValue(arr) {
            //將變數先歸零。
            let max = 0;

            //迴圈取得 array 的內容。
            for (let i = 0; i < arr.length; i++) {
                //若 arr[i] 的值大於上方變數，變數會重新賦值，若小於則不會有改變。
                if (arr[i] > max) {
                    max = arr[i];
                }
            }
            return max;
        }
