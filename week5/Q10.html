第十題：生成費波那契數列

請製作一個函式可以輸入要產生多少個費撥那契數列數字，並回傳對應的費撥那契數列為陣列。

// Todo 待處理
function generateFibonacci(max) {    
}
console.log(generateFibonacci(5))

// Ouput 輸出
[0,1,1,2,3]

回答:

        function generateFibonacci(max) {
            let arr = [0, 1];
            //因為有第一項與第二項，所以起始索引位置為 2 。
            for (let i = 2; i <= max; i++) {
                //之前的兩數相加得到新的係數。
                let fib = arr[i - 1] + arr[i - 2];
                //將取得的係數 push 到 array 中。
                arr.push(fib);
            }
            return arr;
        }      
        console.log(generateFibonacci(5));
