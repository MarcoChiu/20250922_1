題目九：擷取字串內容

// input 輸入
const input = "Hello"
getCharacter(input, 1)
getCharacter(input, 3)
getCharacter(input, 10)

// output 輸出
"H"
"l"
false

回答:

        function getCharacter(str, character) {
            if (str.length > 0) {
                if (parseInt(character) > str.length) {
                    console.log(false);
                } else {
                    console.log(str[character - 1]);
                }
            }
            else {
                console.log('無帶入文字');
            }
        }
        const input = "Hello"
        getCharacter(input, 1);
        getCharacter(input, 3);
        getCharacter(input, 10);
